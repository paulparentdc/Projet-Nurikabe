def precedent!
    case(etat)
    when BLANC
        @etat = POINT
    when NOIR
        @etat = BLANC
    when POINT
        @etat = NOIR
    else
        @etat = BLANC
    end
end

def suivant!
    case(@etat)
    when BLANC
        @etat = NOIR
    when NOIR
        @etat = POINT
    when POINT
        @etat = BLANC
    else
        @etat = BLANC
    end
end
